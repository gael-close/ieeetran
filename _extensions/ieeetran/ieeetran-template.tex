% My tables
\newenvironment{mytable}[1][htbp]{
    \begin{figure}[#1]
    \footnotesize
    \onecolumn
    \begin{minipage}{0.5\textwidth}}
{
    \end{minipage}
    \twocolumn
    \end{figure}}

\newenvironment{mywidetable}[1][tbp]{
    \begin{figure*}[#1]
    \footnotesize
    \onecolumn
    \begin{minipage}{1.0\textwidth}}
{
    \end{minipage}
    \twocolumn
    \end{figure*}}

\renewenvironment{table*}{\begin{mywidetable}}{\end{mywidetable}\ignorespacesafterend}
